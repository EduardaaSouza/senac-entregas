function imprimirApelidos (objeto) {
       


// Criar um objeto
const pessoa = {
    nome: 'Vitor Hugo',
    apelidos: [ 'Vitinho', 'Vitão', 'Vit' ],
}

// Criar um novo objeto com spread substituindo os apelidos
const novosApelidos = {
    ...pessoa,
    apelidos: [ 'Vits', 'Vi', 'Vitu' ],
}

// Declarar a função
imprimirApelidos(pessoa)

}
